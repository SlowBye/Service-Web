import express from 'express';
import jwt from 'jsonwebtoken';

const router = express.Router();
const validTokens = []; // Liste des tokens valides, à stocker dans une base de données en production

/**
 * Fonction permettant de déconnecter un utilisateur.
 * @name disconnect
 * @function
 * @param {Object} req - Requête envoyée par le client.
 * @param {Object} res - Réponse envoyée au client.
 * @returns {Object} - Résultat de la requête.
 */
router.post('/', async (req, res) => {
    const authHeader = req.headers['authorization'];

    if (!authHeader || !authHeader.startsWith('Bearer ')) {
        return res.status(400).send('Invalid token format');
    }

    const token = authHeader.split(' ')[1];

    jwt.verify(token, 'secret', (err, decoded) => {
        if (err) {
            return res.status(401).send('Invalid token');
        } else {
            const tokenIndex = validTokens.indexOf(token);

            if (tokenIndex !== -1) {
                validTokens.splice(tokenIndex, 1);
                return res.status(200).send('Successfully disconnected');
            } else {
                return res.status(404).send('Token not found');
            }
        }
    });
});

export default router;
